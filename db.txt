CREATE TABLE Restaurantes (
  id SERIAL PRIMARY KEY,
  name varchar(40) UNIQUE NOT NULL,
  link varchar(40) UNIQUE NOT NULL,
  image varchar(255) NOT NULL
);

CREATE TABLE Languages (
  id varchar(2) PRIMARY KEY,
  name varchar(40) UNIQUE NOT NULL
);

CREATE TABLE Rest_Languages (
  id SERIAL PRIMARY KEY,
  restaurante_id INTEGER NOT NULL,
  language_id varchar(2),
  FOREIGN KEY (restaurante_id) REFERENCES Restaurantes(id),
  FOREIGN KEY (language_id) REFERENCES Languages(id),
  UNIQUE (restaurante_id, language_id)
);

CREATE TABLE Categories (
  id SERIAL PRIMARY KEY,
  rest_language_id INTEGER NOT NULL,
  name varchar(40),
  link varchar(40),
  parent_id INTEGER,
  image varchar(255),
  FOREIGN KEY (rest_language_id) REFERENCES Rest_Languages(id),
  FOREIGN KEY (parent_id) REFERENCES Categories(id),
  UNIQUE (rest_language_id, link)
);
CREATE INDEX idx_categories_parent_id ON Categories(parent_id);

CREATE TABLE Dishes (
  id SERIAL PRIMARY KEY,
  category_id INTEGER NOT NULL,
  name VARCHAR(100) NOT NULL,
  image VARCHAR(255),
  link VARCHAR(100),
  description VARCHAR(255),
  price INTEGER,
  FOREIGN KEY (category_id) REFERENCES Categories(id),
  UNIQUE (category_id, link)
);
CREATE INDEX idx_dishes_category_id ON Dishes(category_id);
CREATE INDEX idx_dishes_price ON Dishes(price);
